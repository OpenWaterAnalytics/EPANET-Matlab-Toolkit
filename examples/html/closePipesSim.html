while (tstep>0)
    t=d.runHydraulicAnalysis;
    d.setLinkStatus(index ,Status(i)); i=i+1;
    F=[F; d.getLinkFlows];
    tstep=d.nextHydraulicAnalysisStep;
end

d.closeHydraulicAnalysis;
Flows = F(:,index);
T = table(Flows,Status);
